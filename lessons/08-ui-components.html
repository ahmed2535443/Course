🔧 الخطوة 1: دالة cn() المساعدة

دالة cn() هي دالة مهمة لدمج فئات Tailwind بشكل ذكي. هنستخدم مكتبة clsx و tailwind-merge:

lib/cn.js javascript 
import { clsx } from 'clsx';
import { twMerge } from 'tailwind-merge';

// دالة لدمج فئات CSS بشكل ذكي
export function cn(...inputs) {
  return twMerge(clsx(inputs));
}

// أمثلة على الاستخدام:
// cn("px-4 py-2", isActive && "bg-blue-500")
// cn("text-red-500", className) // يمكن تجاوز الألوان
💡 لماذا نحتاج cn()؟

دالة cn() تحل مشكلتين:

  • دمج الفئات المتشابهة: مثلاً cn("px-4", "px-8") يعطيك px-8 فقط
  • الشروط المنطقية: تسمح بإضافة فئات بناءً على شروط
  • تخصيص المكونات: تسمح للمستخدم بتمرير فئات إضافية

🎨 الخطوة 2: مكون Button

مكون Button هو من أهم المكونات. هنشئه بعدة أشكال:

components/ui/Button.jsx jsx 
"use client";

import { cn } from '@/lib/cn';
import { Loader2 } from 'lucide-react';

// أنواع الـ Button
const buttonVariants = {
  primary: "bg-gradient-to-r from-indigo-600 to-purple-600 text-white hover:from-indigo-700 hover:to-purple-700 shadow-lg shadow-indigo-500/25",
  secondary: "bg-white/10 text-white border border-white/20 hover:bg-white/20 backdrop-blur-sm",
  ghost: "text-white/80 hover:text-white hover:bg-white/10",
  danger: "bg-red-500 text-white hover:bg-red-600 shadow-lg shadow-red-500/25",
  outline: "border-2 border-indigo-500 text-indigo-500 hover:bg-indigo-500 hover:text-white",
};

// أحجام الـ Button
const buttonSizes = {
  sm: "px-3 py-1.5 text-sm",
  md: "px-4 py-2 text-base",
  lg: "px-6 py-3 text-lg",
  xl: "px-8 py-4 text-xl",
};

export default function Button({
  children,
  variant = "primary",
  size = "md",
  isLoading = false,
  disabled = false,
  className,
  ...props
}) {
  return (
    <button
      className={cn(
        // الفئات الأساسية
        "inline-flex items-center justify-center rounded-xl font-semibold transition-all duration-300",
        "focus:outline-none focus:ring-2 focus:ring-indigo-500 focus:ring-offset-2",
        "disabled:opacity-50 disabled:cursor-not-allowed",
        // النوع
        buttonVariants[variant],
        // الحجم
        buttonSizes[size],
        // الفئات الإضافية
        className
      )}
      disabled={disabled || isLoading}
      {...props}
    >
      {isLoading && (
        <Loader2 className="w-4 h-4 ml-2 animate-spin" />
      )}
      {children}
    </button>
  );
}

📝 الخطوة 3: مكون Input

مكون Input مع Label وحالات الخطأ:

components/ui/Input.jsx jsx 
"use client";

import { forwardRef } from 'react';
import { cn } from '@/lib/cn';

const Input = forwardRef(({
  label,
  error,
  className,
  type = "text",
  ...props
}, ref) => {
  return (
    <div className="w-full">
      {/* Label */}
      {label && (
        <label className="block text-sm font-medium text-gray-200 mb-2">
          {label}
        </label>
      )}
      
      {/* Input */}
      <input
        ref={ref}
        type={type}
        className={cn(
          "w-full px-4 py-3 rounded-xl bg-white/5 border text-white",
          "placeholder:text-gray-400",
          "focus:outline-none focus:ring-2 focus:ring-indigo-500",
          "transition-all duration-300",
          // حالة الخطأ
          error
            ? "border-red-500 focus:ring-red-500"
            : "border-white/10 hover:border-white/20",
          className
        )}
        {...props}
      />
      
      {/* رسالة الخطأ */}
      {error && (
        <p className="mt-2 text-sm text-red-400 flex items-center gap-2">
          <svg className="w-4 h-4" fill="currentColor" viewBox="0 0 20 20">
            <path fillRule="evenodd" d="M18 10a8 8 0 11-16 0 8 8 0 0116 0zm-7 4a1 1 0 11-2 0 1 1 0 012 0zm-1-9a1 1 0 00-1 1v4a1 1 0 102 0V6a1 1 0 00-1-1z" clipRule="evenodd" />
          </svg>
          {error}
        </p>
      )}
    </div>
  );
});

Input.displayName = 'Input';

export default Input;

🃏 الخطوة 4: مكون Card

مكون Card بتأثير Glass للعروض والمنتجات:

components/ui/Card.jsx jsx 
import { cn } from '@/lib/cn';

const cardVariants = {
  default: "bg-white/5 backdrop-blur-lg border border-white/10",
  glass: "bg-white/10 backdrop-blur-xl border border-white/20",
  solid: "bg-gray-800 border border-gray-700",
  gradient: "bg-gradient-to-br from-white/10 to-white/5 backdrop-blur-lg border border-white/20",
};

export default function Card({
  children,
  variant = "default",
  hover = true,
  className,
  ...props
}) {
  return (
    <div
      className={cn(
        // الفئات الأساسية
        "rounded-2xl p-6 shadow-xl",
        // النوع
        cardVariants[variant],
        // تأثير Hover
        hover && "hover:scale-105 hover:shadow-2xl transition-all duration-300",
        // الفئات الإضافية
        className
      )}
      {...props}
    >
      {children}
    </div>
  );
}

// مكون Card.Header
Card.Header = function CardHeader({ children, className }) {
  return (
    <div className={cn("mb-4", className)}>
      {children}
    </div>
  );
};

// مكون Card.Content
Card.Content = function CardContent({ children, className }) {
  return (
    <div className={cn("", className)}>
      {children}
    </div>
  );
};

// مكون Card.Footer
Card.Footer = function CardFooter({ children, className }) {
  return (
    <div className={cn("mt-4 pt-4 border-t border-white/10", className)}>
      {children}
    </div>
  );
};

🏷️ الخطوة 5: مكون Badge

مكون Badge للتصنيفات والحالات:

components/ui/Badge.jsx jsx 
import { cn } from '@/lib/cn';

const badgeVariants = {
  default: "bg-white/10 text-white",
  primary: "bg-indigo-500/20 text-indigo-300 border border-indigo-500/30",
  success: "bg-green-500/20 text-green-300 border border-green-500/30",
  warning: "bg-yellow-500/20 text-yellow-300 border border-yellow-500/30",
  danger: "bg-red-500/20 text-red-300 border border-red-500/30",
  info: "bg-blue-500/20 text-blue-300 border border-blue-500/30",
};

const badgeSizes = {
  sm: "px-2 py-0.5 text-xs",
  md: "px-3 py-1 text-sm",
  lg: "px-4 py-1.5 text-base",
};

export default function Badge({
  children,
  variant = "default",
  size = "md",
  dot = false,
  className,
  ...props
}) {
  return (
    <span
      className={cn(
        "inline-flex items-center gap-1.5 rounded-full font-medium",
        badgeVariants[variant],
        badgeSizes[size],
        className
      )}
      {...props}
    >
      {dot && (
        <span className={cn(
          "w-1.5 h-1.5 rounded-full",
          variant === "success" && "bg-green-400",
          variant === "warning" && "bg-yellow-400",
          variant === "danger" && "bg-red-400",
          variant === "info" && "bg-blue-400",
          variant === "primary" && "bg-indigo-400",
          variant === "default" && "bg-white"
        )} />
      )}
      {children}
    </span>
  );
}

🔔 الخطوة 6: مكون Toast

مكون Toast للإشعارات:

components/ui/Toast.jsx jsx 
"use client";

import { useState, useEffect, createContext, useContext } from 'react';
import { cn } from '@/lib/cn';
import { X, CheckCircle, AlertCircle, Info, AlertTriangle } from 'lucide-react';

const ToastContext = createContext(null);

// أنواع Toast
const toastTypes = {
  success: {
    icon: CheckCircle,
    className: "bg-green-500/20 border-green-500/30 text-green-300",
  },
  error: {
    icon: AlertCircle,
    className: "bg-red-500/20 border-red-500/30 text-red-300",
  },
  warning: {
    icon: AlertTriangle,
    className: "bg-yellow-500/20 border-yellow-500/30 text-yellow-300",
  },
  info: {
    icon: Info,
    className: "bg-blue-500/20 border-blue-500/30 text-blue-300",
  },
};

// مزود Toast
export function ToastProvider({ children }) {
  const [toasts, setToasts] = useState([]);

  const addToast = (message, type = "info", duration = 3000) => {
    const id = Date.now();
    setToasts(prev => [...prev, { id, message, type }]);
    
    setTimeout(() => {
      removeToast(id);
    }, duration);
  };

  const removeToast = (id) => {
    setToasts(prev => prev.filter(toast => toast.id !== id));
  };

  return (
    <ToastContext.Provider value={{ addToast, removeToast }}>
      {children}
      
      {/* Toast Container */}
      <div className="fixed bottom-4 right-4 z-50 flex flex-col gap-2">
        {toasts.map(toast => (
          <Toast
            key={toast.id}
            message={toast.message}
            type={toast.type}
            onClose={() => removeToast(toast.id)}
          />
        ))}
      </div>
    </ToastContext.Provider>
  );
}

// Hook لاستخدام Toast
export function useToast() {
  const context = useContext(ToastContext);
  if (!context) {
    throw new Error("useToast must be used within ToastProvider");
  }
  return context;
}

// مكون Toast
function Toast({ message, type = "info", onClose }) {
  const { icon: Icon, className } = toastTypes[type];

  return (
    <div
      className={cn(
        "flex items-center gap-3 px-4 py-3 rounded-xl border backdrop-blur-lg shadow-xl animate-slide-in",
        className
      )}
    >
      <Icon className="w-5 h-5 flex-shrink-0" />
      <p className="flex-1 text-sm font-medium">{message}</p>
      <button
        onClick={onClose}
        className="p-1 hover:bg-white/10 rounded-lg transition-colors"
      >
        <X className="w-4 h-4" />
      </button>
    </div>
  );
}

استخدام Toast:

مثال الاستخدام jsx 
"use client";

import { useToast } from '@/components/ui/Toast';
import Button from '@/components/ui/Button';

export default function MyComponent() {
  const { addToast } = useToast();

  const handleSuccess = () => {
    addToast("تمت الإضافة بنجاح!", "success");
  };

  const handleError = () => {
    addToast("حدث خطأ، حاول مرة أخرى", "error");
  };

  return (
    <div>
      <Button onClick={handleSuccess}>إضافة ناجحة</Button>
      <Button onClick={handleError}>إضافة فاشلة</Button>
    </div>
  );
}

📦 الخطوة 7: مكون Container

مكون Container لتخطيط المحتوى:

components/ui/Container.jsx jsx 
import { cn } from '@/lib/cn';

const containerVariants = {
  default: "max-w-7xl mx-auto px-4 sm:px-6 lg:px-8",
  narrow: "max-w-4xl mx-auto px-4 sm:px-6 lg:px-8",
  wide: "max-w-full mx-auto px-4 sm:px-6 lg:px-8",
  fluid: "w-full px-4 sm:px-6 lg:px-8",
};

export default function Container({
  children,
  variant = "default",
  as: Component = "div",
  className,
  ...props
}) {
  return (
    <Component
      className={cn(containerVariants[variant], className)}
      {...props}
    >
      {children}
    </Component>
  );
}
