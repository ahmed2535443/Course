✨ الخطوة 8: استخدام حالات التحميل

components/products/ProductGrid.jsx jsx 
"use client";

import { useState, useEffect } from 'react';
import ProductCard from './ProductCard';
import ProductCardSkeleton from '@/components/ui/ProductCardSkeleton';
import { motion, AnimatePresence } from 'framer-motion';

export default function ProductGrid({ initialProducts = [], filters }) {
  const [products, setProducts] = useState(initialProducts);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetchProducts();
  }, [filters]);

  const fetchProducts = async () => {
    setLoading(true);
    setError(null);

    try {
      const params = new URLSearchParams();
      if (filters?.category) params.set('category', filters.category);
      if (filters?.minPrice) params.set('minPrice', filters.minPrice);
      if (filters?.maxPrice) params.set('maxPrice', filters.maxPrice);
      if (filters?.sort) params.set('sort', filters.sort);

      const response = await fetch(`/api/products?${params.toString()}`);
      
      if (!response.ok) {
        throw new Error('فشل في جلب المنتجات');
      }

      const data = await response.json();
      setProducts(data);
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  if (error) {
    return (
      

        
{error}

        
      

    );
  }

  return (
    

      
        {loading ? (
          // Show skeletons while loading
          [...Array(8)].map((_, i) => (
            
              
            
          ))
        ) : (
          // Show actual products
          products.map((product, index) => (
            
              
            
          ))
        )}
      
    

  );
}
💡 نصائح لتحسين تجربة التحميل
  • استخدم Suspense مع loading.js لتحميل الأجزاء البطيئة
  • أضف تأخير صغير (300ms) قبل إظهار الـ Skeleton لتجنب الوميض
  • استخدم staggerChildren في Framer Motion لتحريك العناصر تدريجياً
  • حافظ على حجم الـ Skeleton نفس حجم المحتوى الفعلي
