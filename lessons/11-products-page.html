🛍️ الخطوة 1: صفحة المنتجات الرئيسية

app/(shop)/products/page.js jsx 
"use client";

import { useState, useEffect } from 'react';
import { useSearchParams } from 'next/navigation';
import ProductGrid from '@/components/products/ProductGrid';
import SearchBar from '@/components/products/SearchBar';
import CategoryFilter from '@/components/products/CategoryFilter';
import SortingOptions from '@/components/products/SortingOptions';
import Pagination from '@/components/products/Pagination';
import Container from '@/components/ui/Container';

export default function ProductsPage() {
  const searchParams = useSearchParams();
  
  const [products, setProducts] = useState([]);
  const [loading, setLoading] = useState(true);
  const [totalPages, setTotalPages] = useState(1);
  const [currentPage, setCurrentPage] = useState(1);
  
  // الفلاتر
  const [filters, setFilters] = useState({
    search: searchParams.get('search') || '',
    category: searchParams.get('category') || '',
    sort: searchParams.get('sort') || 'newest',
    page: parseInt(searchParams.get('page')) || 1,
  });

  // جلب المنتجات
  useEffect(() => {
    fetchProducts();
  }, [filters]);

  const fetchProducts = async () => {
    setLoading(true);
    try {
      const params = new URLSearchParams();
      if (filters.search) params.set('search', filters.search);
      if (filters.category) params.set('category', filters.category);
      params.set('sort', filters.sort);
      params.set('page', filters.page.toString());
      params.set('limit', '12');

      const response = await fetch(`/api/products?${params.toString()}`);
      const data = await response.json();
      
      setProducts(data.products);
      setTotalPages(data.totalPages);
      setCurrentPage(data.currentPage);
    } catch (error) {
      console.error('Error fetching products:', error);
    } finally {
      setLoading(false);
    }
  };

  // تحديث الفلاتر
  const updateFilters = (newFilters) => {
    setFilters(prev => ({ ...prev, ...newFilters, page: 1 }));
  };

  return (
    <Container className="py-8">
      {/* Header */}
      <div className="mb-8">
        <h1 className="text-3xl font-bold text-white mb-2">
          جميع المنتجات
        </h1>
        <p className="text-gray-400">
          اكتشف مجموعتنا الواسعة من المنتجات عالية الجودة
        </p>
      </div>

      {/* Search & Filters */}
      <div className="flex flex-col lg:flex-row gap-4 mb-8">
        {/* Search */}
        <div className="flex-1">
          <SearchBar
            value={filters.search}
            onChange={(search) => updateFilters({ search })}
          />
        </div>

        {/* Sorting */}
        <SortingOptions
          value={filters.sort}
          onChange={(sort) => updateFilters({ sort })}
        />
      </div>

      <div className="flex flex-col lg:flex-row gap-8">
        {/* Sidebar Filters */}
        <aside className="lg:w-64 flex-shrink-0">
          <CategoryFilter
            selectedCategory={filters.category}
            onChange={(category) => updateFilters({ category })}
          />
        </aside>

        {/* Products Grid */}
        <div className="flex-1">
          <ProductGrid
            products={products}
            loading={loading}
          />

          {/* Pagination */}
          {totalPages > 1 && (
            <div className="mt-8">
              <Pagination
                currentPage={currentPage}
                totalPages={totalPages}
                onPageChange={(page) => updateFilters({ page })}
              />
            </div>
          )}
        </div>
      </div>
    </Container>
  );
}

🔍 الخطوة 2: شريط البحث

components/products/SearchBar.jsx jsx 
"use client";

import { useState, useEffect } from 'react';
import { Search, X } from 'lucide-react';
import { cn } from '@/lib/cn';

export default function SearchBar({ value, onChange }) {
  const [localValue, setLocalValue] = useState(value);

  // Debounce للبحث
  useEffect(() => {
    const timer = setTimeout(() => {
      onChange(localValue);
    }, 300);

    return () => clearTimeout(timer);
  }, [localValue, onChange]);

  return (
    <div className="relative">
      {/* Search Icon */}
      <Search className="absolute right-4 top-1/2 -translate-y-1/2 w-5 h-5 text-gray-400" />

      {/* Input */}
      <input
        type="text"
        value={localValue}
        onChange={(e) => setLocalValue(e.target.value)}
        placeholder="ابحث عن منتجات..."
        className={cn(
          "w-full pr-12 pl-12 py-3 rounded-xl",
          "bg-white/5 border border-white/10",
          "text-white placeholder:text-gray-400",
          "focus:outline-none focus:ring-2 focus:ring-indigo-500",
          "transition-all duration-300"
        )}
      />

      {/* Clear Button */}
      {localValue && (
        <button
          onClick={() => {
            setLocalValue('');
            onChange('');
          }}
          className="absolute left-4 top-1/2 -translate-y-1/2 p-1 rounded-lg text-gray-400 hover:text-white hover:bg-white/10 transition-colors"
        >
          <X className="w-4 h-4" />
        </button>
      )}
    </div>
  );
}
💡 ما هو Debounce؟

Deounce هو تقنية تؤخر تنفيذ الدالة حتى يتوقف المستخدم عن الكتابة لفترة قصيرة (300ms هنا). هذا يمنع إرسال طلب لكل حرف يكتبه المستخدم.

📂 الخطوة 3: فلتر التصنيفات

components/products/CategoryFilter.jsx jsx 
"use client";

import { useState, useEffect } from 'react';
import { cn } from '@/lib/cn';
import { ChevronDown, ChevronUp } from 'lucide-react';

export default function CategoryFilter({ selectedCategory, onChange }) {
  const [categories, setCategories] = useState([]);
  const [isExpanded, setIsExpanded] = useState(true);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchCategories();
  }, []);

  const fetchCategories = async () => {
    try {
      const response = await fetch('/api/categories');
      const data = await response.json();
      setCategories(data);
    } catch (error) {
      console.error('Error fetching categories:', error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="bg-white/5 backdrop-blur-lg rounded-2xl border border-white/10 p-4">
      {/* Header */}
      <button
        onClick={() => setIsExpanded(!isExpanded)}
        className="w-full flex items-center justify-between mb-4"
      >
        <h3 className="text-white font-semibold">التصنيفات</h3>
        {isExpanded ? (
          <ChevronUp className="w-5 h-5 text-gray-400" />
        ) : (
          <ChevronDown className="w-5 h-5 text-gray-400" />
        )}
      </button>

      {/* Categories List */}
      {isExpanded && (
        <div className="space-y-2">
          {/* All Categories */}
          <button
            onClick={() => onChange('')}
            className={cn(
              "w-full text-right px-3 py-2 rounded-lg text-sm transition-colors",
              !selectedCategory
                ? "bg-indigo-500/20 text-indigo-300"
                : "text-gray-400 hover:text-white hover:bg-white/5"
            )}
          >
            الكل
          </button>

          {loading ? (
            // Loading Skeleton
            [...Array(6)].map((_, i) => (
              <div
                key={i}
                className="h-10 bg-white/5 rounded-lg animate-pulse"
              />
            ))
          ) : (
            categories.map((category) => (
              <button
                key={category.id}
                onClick={() => onChange(category.slug)}
                className={cn(
                  "w-full text-right px-3 py-2 rounded-lg text-sm transition-colors",
                  selectedCategory === category.slug
                    ? "bg-indigo-500/20 text-indigo-300"
                    : "text-gray-400 hover:text-white hover:bg-white/5"
                )}
              >
                <div className="flex items-center justify-between">
                  <span>{category.name}</span>
                  <span className="text-xs text-gray-500">
                    ({category.productCount})
                  </span>
                </div>
              </button>
            ))
          )}
        </div>
      )}
    </div>
  );
}

↕️ الخطوة 4: خيارات الترتيب

components/products/SortingOptions.jsx jsx 
"use client";

import { useState } from 'react';
import { ArrowUpDown, ChevronDown } from 'lucide-react';
import { cn } from '@/lib/cn';

const sortOptions = [
  { value: 'newest', label: 'الأحدث' },
  { value: 'price_asc', label: 'السعر: من الأقل للأعلى' },
  { value: 'price_desc', label: 'السعر: من الأعلى للأقل' },
  { value: 'rating', label: 'الأعلى تقييماً' },
  { value: 'popular', label: 'الأكثر مبيعاً' },
];

export default function SortingOptions({ value, onChange }) {
  const [isOpen, setIsOpen] = useState(false);

  const selectedOption = sortOptions.find(opt => opt.value === value);

  return (
    <div className="relative">
      {/* Trigger */}
      <button
        onClick={() => setIsOpen(!isOpen)}
        className={cn(
          "flex items-center gap-2 px-4 py-3 rounded-xl",
          "bg-white/5 border border-white/10",
          "text-white hover:bg-white/10 transition-colors",
          isOpen && "ring-2 ring-indigo-500"
        )}
      >
        <ArrowUpDown className="w-5 h-5 text-gray-400" />
        <span>{selectedOption?.label || 'ترتيب حسب'}</span>
        <ChevronDown className={cn(
          "w-4 h-4 text-gray-400 transition-transform",
          isOpen && "rotate-180"
        )} />
      </button>

      {/* Dropdown */}
      {isOpen && (
        <>
          {/* Overlay */}
          <div
            className="fixed inset-0 z-10"
            onClick={() => setIsOpen(false)}
          />
          
          {/* Options */}
          <div className="absolute top-full left-0 mt-2 w-64 bg-gray-800 border border-white/10 rounded-xl shadow-xl z-20 overflow-hidden">
            {sortOptions.map((option) => (
              <button
                key={option.value}
                onClick={() => {
                  onChange(option.value);
                  setIsOpen(false);
                }}
                className={cn(
                  "w-full text-right px-4 py-3 text-sm transition-colors",
                  value === option.value
                    ? "bg-indigo-500/20 text-indigo-300"
                    : "text-gray-300 hover:bg-white/5 hover:text-white"
                )}
              >
                {option.label}
              </button>
            ))}
          </div>
        </>
      )}
    </div>
  );
}

📄 الخطوة 5: ترقيم الصفحات

components/products/Pagination.jsx jsx 
"use client";

import { ChevronRight, ChevronLeft, MoreHorizontal } from 'lucide-react';
import { cn } from '@/lib/cn';

export default function Pagination({ currentPage, totalPages, onPageChange }) {
  // حساب الصفحات المرئية
  const getPageNumbers = () => {
    const pages = [];
    const showEllipsis = totalPages > 7;

    if (!showEllipsis) {
      for (let i = 1; i <= totalPages; i++) {
        pages.push(i);
      }
    } else {
      // الصفحة الأولى
      pages.push(1);

      if (currentPage > 3) {
        pages.push('...');
      }

      // الصفحات الوسطى
      for (
        let i = Math.max(2, currentPage - 1);
        i <= Math.min(totalPages - 1, currentPage + 1);
        i++
      ) {
        pages.push(i);
      }

      if (currentPage < totalPages - 2) {
        pages.push('...');
      }

      // الصفحة الأخيرة
      pages.push(totalPages);
    }

    return pages;
  };

  return (
    <div className="flex items-center justify-center gap-2">
      {/* Previous Button */}
      <button
        onClick={() => onPageChange(currentPage - 1)}
        disabled={currentPage === 1}
        className={cn(
          "p-2 rounded-lg transition-colors",
          currentPage === 1
            ? "text-gray-600 cursor-not-allowed"
            : "text-gray-400 hover:text-white hover:bg-white/10"
        )}
      >
        <ChevronRight className="w-5 h-5" />
      </button>

      {/* Page Numbers */}
      {getPageNumbers().map((page, index) => (
        <button
          key={index}
          onClick={() => typeof page === 'number' && onPageChange(page)}
          disabled={page === '...'}
          className={cn(
            "min-w-[40px] h-10 rounded-lg text-sm font-medium transition-colors",
            page === currentPage
              ? "bg-indigo-500 text-white"
              : page === '...'
              ? "text-gray-600 cursor-default"
              : "text-gray-400 hover:text-white hover:bg-white/10"
          )}
        >
          {page === '...' ? (
            <MoreHorizontal className="w-5 h-5 mx-auto" />
          ) : (
            page
          )}
        </button>
      ))}

      {/* Next Button */}
      <button
        onClick={() => onPageChange(currentPage + 1)}
        disabled={currentPage === totalPages}
        className={cn(
          "p-2 rounded-lg transition-colors",
          currentPage === totalPages
            ? "text-gray-600 cursor-not-allowed"
            : "text-gray-400 hover:text-white hover:bg-white/10"
        )}
      >
        <ChevronLeft className="w-5 h-5" />
      </button>
    </div>
  );
}

import ProductCard from './ProductCard';

export default function ProductGrid({ products, loading }) {
  // حالة التحميل
  if (loading) {
    return (
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-6">
        {[...Array(12)].map((_, i) => (
          <div
            key={i}
            className="bg-white/5 rounded-2xl overflow-hidden animate-pulse"
          >
            <div className="aspect-square bg-white/10" />
            <div className="p-4 space-y-3">
              <div className="h-4 bg-white/10 rounded w-1/3" />
              <div className="h-5 bg-white/10 rounded w-2/3" />
              <div className="h-4 bg-white/10 rounded w-1/2" />
              <div className="h-6 bg-white/10 rounded w-1/4" />
            </div>
          </div>
        ))}
      </div>
    );
  }

  // لا توجد منتجات
  if (!products || products.length === 0) {
    return (
      <div className="text-center py-16">
        <div className="w-24 h-24 mx-auto mb-6 rounded-full bg-white/5 flex items-center justify-center">
          <span className="text-4xl">📦</span>
        </div>
        <h3 className="text-xl font-semibold text-white mb-2">
          لا توجد منتجات
        </h3>
        <p className="text-gray-400">
          جرّب تغيير معايير البحث أو الفلتر
        </p>
      </div>
    );
  }

  // عرض المنتجات
  return (
    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-6">
      {products.map((product) => (
        <ProductCard key={product.id} product={product} />
      ))}
    </div>
  );
}

🔗 الخطوة 7: حاوية الفلاتر

components/products/ProductFilters.jsx jsx 
"use client";

import { useState } from 'react';
import { Filter, X } from 'lucide-react';
import { cn } from '@/lib/cn';
import CategoryFilter from './CategoryFilter';

const priceRanges = [
  { id: 'all', label: 'الكل', min: 0, max: Infinity },
  { id: '0-50', label: 'أقل من $50', min: 0, max: 50 },
  { id: '50-100', label: '$50 - $100', min: 50, max: 100 },
  { id: '100-200', label: '$100 - $200', min: 100, max: 200 },
  { id: '200+', label: 'أكثر من $200', min: 200, max: Infinity },
];

export default function ProductFilters({ filters, onFilterChange }) {
  const [isMobileOpen, setIsMobileOpen] = useState(false);

  const FilterContent = () => (
    <div className="space-y-6">
      {/* Price Range */}
      <div>
        <h3 className="text-white font-semibold mb-4">نطاق السعر</h3>
        <div className="space-y-2">
          {priceRanges.map((range) => (
            <button
              key={range.id}
              onClick={() => onFilterChange({ priceRange: range.id })}
              className={cn(
                "w-full text-right px-3 py-2 rounded-lg text-sm transition-colors",
                filters.priceRange === range.id
                  ? "bg-indigo-500/20 text-indigo-300"
                  : "text-gray-400 hover:text-white hover:bg-white/5"
              )}
            >
              {range.label}
            </button>
          ))}
        </div>
      </div>

      {/* Rating Filter */}
      <div>
        <h3 className="text-white font-semibold mb-4">التقييم</h3>
        <div className="space-y-2">
          {[5, 4, 3, 2, 1].map((rating) => (
            <button
              key={rating}
              onClick={() => onFilterChange({ minRating: rating })}
              className={cn(
                "w-full text-right px-3 py-2 rounded-lg text-sm transition-colors flex items-center gap-2",
                filters.minRating === rating
                  ? "bg-indigo-500/20 text-indigo-300"
                  : "text-gray-400 hover:text-white hover:bg-white/5"
              )}
            >
              <div className="flex">
                {[...Array(5)].map((_, i) => (
                  <span
                    key={i}
                    className={cn(
                      "text-lg",
                      i < rating ? "text-yellow-400" : "text-gray-600"
                    )}
                  >
                    ★
                  </span>
                ))}
              </div>
              <span>وأعلى</span>
            </button>
          ))}
        </div>
      </div>
    </div>
  );

  return (
    <>
      {/* Mobile Filter Button */}
      <button
        onClick={() => setIsMobileOpen(true)}
        className="lg:hidden fixed bottom-4 left-4 z-30 flex items-center gap-2 px-4 py-3 rounded-full bg-indigo-500 text-white shadow-lg"
      >
        <Filter className="w-5 h-5" />
        <span>الفلاتر</span>
      </button>

      {/* Desktop Sidebar */}
      <div className="hidden lg:block">
        <FilterContent />
      </div>

      {/* Mobile Sidebar */}
      {isMobileOpen && (
        <div className="lg:hidden fixed inset-0 z-40">
          <div
            className="absolute inset-0 bg-black/50"
            onClick={() => setIsMobileOpen(false)}
          />
          <div className="absolute right-0 top-0 h-full w-80 bg-gray-900 p-4 overflow-y-auto">
            <div className="flex items-center justify-between mb-6">
              <h2 className="text-white font-bold text-lg">الفلاتر</h2>
              <button
                onClick={() => setIsMobileOpen(false)}
                className="p-2 rounded-lg text-gray-400 hover:text-white hover:bg-white/10"
              >
                <X className="w-5 h-5" />
              </button>
            </div>
            <FilterContent />
          </div>
        </div>
      )}
    </>
  );
}
