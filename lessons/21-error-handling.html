    setToasts((prev) => [...prev, { id, message, type }]);

        removeToast(id);
      }, duration);

🌐 الخطوة 6: معالجة أخطاء API

utils/api.js javascript 
/**
 * دالة موحدة لإجراء استدعاءات API مع معالجة الأخطاء
 */
export async function apiCall(url, options = {}) {
  const {
    method = 'GET',
    headers = {},
    body,
    timeout = 10000,
  } = options;

  // إعداد Headers الافتراضية
  const defaultHeaders = {
    'Content-Type': 'application/json',
  };

  // دمج Headers
  const mergedHeaders = {
    ...defaultHeaders,
    ...headers,
  };

  // إعداد الـ AbortController للـ timeout
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeout);

  try {
    const fetchOptions = {
      method,
      headers: mergedHeaders,
      signal: controller.signal,
    };

    if (body && method !== 'GET') {
      fetchOptions.body = JSON.stringify(body);
    }

    const response = await fetch(url, fetchOptions);
    clearTimeout(timeoutId);

    // التحقق من نوع المحتوى
    const contentType = response.headers.get('content-type');
    let data;

    if (contentType?.includes('application/json')) {
      data = await response.json();
    } else {
      data = await response.text();
    }

    // التحقق من نجاح الطلب
    if (!response.ok) {
      const error = new Error(data.message || data.error || 'حدث خطأ في الخادم');
      error.status = response.status;
      error.data = data;
      throw error;
    }

    return { success: true, data };
  } catch (error) {
    clearTimeout(timeoutId);

    // معالجة أنواع الأخطاء المختلفة
    if (error.name === 'AbortError') {
      return {
        success: false,
        error: {
          type: 'timeout',
          message: 'انتهت مهلة الاتصال بالخادم',
        },
      };
    }

    if (!navigator.onLine) {
      return {
        success: false,
        error: {
          type: 'network',
          message: 'لا يوجد اتصال بالإنترنت',
        },
      };
    }

    return {
      success: false,
      error: {
        type: error.status ? 'api' : 'unknown',
        status: error.status,
        message: error.message || 'حدث خطأ غير متوقع',
        data: error.data,
      },
    };
  }
}

/**
 * Hook لاستخدام API مع حالة التحميل
 */
export function useApi() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const execute = async (url, options) => {
    setLoading(true);
    setError(null);

    const result = await apiCall(url, options);

    if (!result.success) {
      setError(result.error);
    }

    setLoading(false);
    return result;
  };

  const clearError = () => setError(null);

  return { loading, error, execute, clearError };
}

"use client";

import { useCallback } from 'react';
import { useToast } from '@/components/ui/Toast';

/**
 * Hook موحد لمعالجة الأخطاء
 */
export function useErrorHandler() {
  const { error: showError, warning, info } = useToast();

  const handleError = useCallback((error, context = {}) => {
    console.error('Error:', error, context);

    // تحديد نوع الخطأ وعرض رسالة مناسبة
    if (error?.status === 401) {
      showError('يرجى تسجيل الدخول أولاً', 5000);
      // يمكنك إعادة التوجيه لصفحة تسجيل الدخول
      // router.push('/sign-in');
    } else if (error?.status === 403) {
      showError('ليس لديك صلاحية للقيام بهذه العملية', 5000);
    } else if (error?.status === 404) {
      warning('العنصر المطلوب غير موجود', 5000);
    } else if (error?.status === 422) {
      // أخطاء التحقق من صحة البيانات
      const validationErrors = error.data?.errors;
      if (validationErrors) {
        Object.values(validationErrors).flat().forEach((msg) => {
          showError(msg, 5000);
        });
      } else {
        showError(error.message || 'بيانات غير صالحة', 5000);
      }
    } else if (error?.status === 429) {
      warning('طلبات كثيرة جداً، يرجى الانتظار قليلاً', 5000);
    } else if (error?.status >= 500) {
      showError('خطأ في الخادم، يرجى المحاولة لاحقاً', 5000);
    } else if (error?.type === 'network') {
      showError('لا يوجد اتصال بالإنترنت', 5000);
    } else if (error?.type === 'timeout') {
      warning('انتهت مهلة الاتصال، يرجى المحاولة مجدداً', 5000);
    } else {
      showError(error?.message || 'حدث خطأ غير متوقع', 5000);
    }

    // يمكنك إرسال الخطأ إلى نظام مراقبة
    // Sentry.captureException(error, { extra: context });
  }, [showError, warning]);

  const handleAsyncError = useCallback(async (asyncFn, context = {}) => {
    try {
      return await asyncFn();
    } catch (error) {
      handleError(error, context);
      throw error;
    }
  }, [handleError]);

  return {
    handleError,
    handleAsyncError,
  };
}

/**
 * Hook للتعامل مع أخطاء النماذج
 */
export function useFormErrors() {
  const { handleError } = useErrorHandler();

  const handleFormError = useCallback((error, setErrors) => {
    if (error?.status === 422 && error.data?.errors) {
      // تحويل أخطاء الـ Validation إلى شكل يفهمه النموذج
      const formErrors = {};
      Object.entries(error.data.errors).forEach(([field, messages]) => {
        formErrors[field] = Array.isArray(messages) ? messages[0] : messages;
      });
      setErrors(formErrors);
    } else {
      handleError(error);
    }
  }, [handleError]);

  return { handleFormError };
}
