💡 مقدمة للدرس

أهلاً بيك في الدرس السابع عشر! في الدرس ده هنبني لوحة تحكم Admin. لوحة التحكم دي هي المكان اللي صاحب المتجر بيشوف فيه كل حاجة: عدد الطلبات، الإيرادات، عدد المنتجات، والمستخدمين.

في الدرس ده هنتعلم إزاي نبني واجهة admin كاملة فيها sidebar للتنقل، وإحصائيات، وطلبات أخيرة. وكمان هنتعلم إزاي نحمي مسارات الـ admin من غير مسؤولين!

🔒 الخطوة 1: حماية مسارات Admin

أولاً محتاجين نحمي مسارات Admin حتى لا يصل إليها أي شخص غير مصرح له. نستخدم Clerk middleware للتحقق من صلاحيات المستخدم:

middleware.js javascript 
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server';

// تحديد المسارات المحمية
const isPublicRoute = createRouteMatcher([
  '/',
  '/products(.*)',
  '/categories(.*)',
  '/api(.*)',
  '/sign-in(.*)',
  '/sign-up(.*)',
]);

// مسارات Admin (تحتاج صلاحيات admin)
const isAdminRoute = createRouteMatcher([
  '/admin(.*)',
]);

export default clerkMiddleware(async (auth, req) => {
  // إذا المسار عام، اسمح بالدخول
  if (isPublicRoute(req)) {
    return;
  }

  // إذا المسار خاص بـ admin
  if (isAdminRoute(req)) {
    const { userId } = await auth();
    
    // التحقق من تسجيل الدخول
    if (!userId) {
      return new Response(null, {
        status: 302,
        headers: {
          Location: '/sign-in',
        },
      });
    }

    // التحقق من صلاحيات admin
    // يمكنك استخدام Clerk metadata أو قاعدة بيانات Supabase
    const user = await auth.getUser(userId);
    if (!user?.publicMetadata?.isAdmin) {
      return new Response(null, {
        status: 302,
        headers: {
          Location: '/',
        },
      });
    }
  }

  // لبقية المسارات، تحقق من تسجيل الدخول
  const { userId } = await auth();
  if (!userId) {
    return new Response(null, {
      status: 302,
      headers: {
        Location: '/sign-in',
      },
    });
  }
});

export const config = {
  matcher: [
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    '/(api|trpc)(.*)',
  ],
};
⚠️ مهمة جداً
تأكد من حماية مسارات Admin في كل مكان. لا تعتمد فقط على إخفاء القوائم - يجب التحقق من الصلاحيات على مستوى الـ API والـ Middleware.

📐 الخطوة 2: تخطيط Admin Dashboard

app/(admin)/layout.js jsx 
"use client";

import { useAuth } from '@clerk/nextjs';
import { useRouter } from 'next/navigation';
import { useEffect, useState } from 'react';
import Sidebar from '@/components/admin/Sidebar';
import AdminHeader from '@/components/admin/AdminHeader';
import { Loader2 } from 'lucide-react';

export default function AdminLayout({ children }) {
  const { isLoaded, isSignedIn, userId } = useAuth();
  const router = useRouter();
  const [isAdmin, setIsAdmin] = useState(false);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (isLoaded) {
      checkAdminStatus();
    }
  }, [isLoaded, userId]);

  const checkAdminStatus = async () => {
    try {
      if (!isSignedIn) {
        router.push('/sign-in');
        return;
      }

      // التحقق من صلاحيات admin
      const response = await fetch('/api/admin/check', {
        method: 'GET',
        headers: {
          'Content-Type': 'application/json',
        },
      });

      if (response.ok) {
        setIsAdmin(true);
      } else {
        router.push('/');
      }
    } catch (error) {
      console.error('Error checking admin status:', error);
      router.push('/');
    } finally {
      setLoading(false);
    }
  };

  if (!isLoaded || loading) {
    return (
      <div className="min-h-screen bg-gray-950 flex items-center justify-center">
        <div className="text-center">
          <Loader2 className="w-12 h-12 text-indigo-500 animate-spin mx-auto mb-4" />
          <p className="text-gray-400">جاري التحقق من الصلاحيات...</p>
        </div>
      </div>
    );
  }

  if (!isAdmin) {
    return null;
  }

  return (
    <div className="min-h-screen bg-gray-950 flex">
      {/* Sidebar */}
      <Sidebar />

      {/* Main Content */}
      <div className="flex-1 flex flex-col">
        {/* Header */}
        <AdminHeader />

        {/* Page Content */}
        <main className="flex-1 p-6 overflow-auto">
          {children}
        </main>
      </div>
    </div>
  );
}

  purple: {
    bg: 'bg-purple-500/20',
    text: 'text-purple-400',
    border: 'border-purple-500/30',
  },
};

export default function StatsCard({
  title,
  value,
  change,
  changeType = 'increase',
  icon: Icon,
  color = 'indigo',
  loading = false,
}) {
  const colors = colorMap[color] || colorMap.indigo;

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      className={cn(
        "bg-white/5 backdrop-blur-lg rounded-2xl border p-6",
        colors.border
      )}
    >
      <div className="flex items-start justify-between">
        <div>
          <p className="text-gray-400 text-sm mb-1">{title}</p>
          {loading ? (
            <div className="w-24 h-8 bg-gray-800 rounded animate-pulse" />
          ) : (
            <p className="text-3xl font-bold text-white">{value}</p>
          )}
          {!loading && change !== undefined && (
            <p className={cn(
              "text-sm mt-2",
              changeType === 'increase' ? 'text-green-400' : 'text-red-400'
            )}>
              {changeType === 'increase' ? '↑' : '↓'} {change}%
              <span className="text-gray-500 mr-2">من الشهر الماضي</span>
            </p>
          )}
        </div>
        <div className={cn(
          "w-14 h-14 rounded-xl flex items-center justify-center",
          colors.bg
        )}>
          {loading ? (
            <div className="w-8 h-8 bg-gray-700 rounded animate-pulse" />
          ) : (
            <Icon className={cn("w-7 h-7", colors.text)} />
          )}
        </div>
      </div>
    </motion.div>
  );
}

🔌 الخطوة 8: API Route للإحصائيات

app/api/admin/stats/route.js javascript 
import { NextResponse } from 'next/server';
import { supabase } from '@/lib/supabase';

export async function GET() {
  try {
    // جلب عدد المنتجات
    const { count: totalProducts } = await supabase
      .from('products')
      .select('*', { count: 'exact', head: true });

    // جلب عدد الطلبات
    const { count: totalOrders } = await supabase
      .from('orders')
      .select('*', { count: 'exact', head: true });

    // جلب إجمالي الإيرادات
    const { data: revenueData } = await supabase
      .from('orders')
      .select('total')
      .eq('payment_status', 'paid');

    const totalRevenue = revenueData?.reduce((sum, order) => sum + order.total, 0) || 0;

    // جلب عدد المستخدمين (من Clerk أو Supabase)
    const { count: totalUsers } = await supabase
      .from('users')
      .select('*', { count: 'exact', head: true });

    // حساب الإحصائيات الشهرية
    const now = new Date();
    const startOfMonth = new Date(now.getFullYear(), now.getMonth(), 1);

    const { data: monthlyOrders } = await supabase
      .from('orders')
      .select('total')
      .gte('created_at', startOfMonth.toISOString())
      .eq('payment_status', 'paid');

    const monthlyRevenue = monthlyOrders?.reduce((sum, order) => sum + order.total, 0) || 0;

    return NextResponse.json({
      totalProducts: totalProducts || 0,
      totalOrders: totalOrders || 0,
      totalRevenue: Math.round(totalRevenue),
      totalUsers: totalUsers || 0,
      monthlyRevenue: Math.round(monthlyRevenue),
      averageOrderValue: totalOrders > 0 
        ? Math.round(totalRevenue / totalOrders) 
        : 0,
    });
  } catch (error) {
    console.error('Error fetching stats:', error);
    return NextResponse.json(
      { error: 'Failed to fetch stats' },
      { status: 500 }
    );
  }
}
