💡 مقدمة للدرس

أهلاً بيك في الدرس التالت عشر! في الدرس ده هنتعلم نبني نظام سلة مشتريات كامل. السلة دي من أهم مكونات أي متجر إلكتروني - بدونها المستخدم مش هيبقى يقدر يشتري أي حاجة!

خلال الدرس ده هنعمل كل حاجة من الصفر: هنعرف إزاي نخزن المنتجات اللي المستخدم اختارها، وإزاي نحسب الإجماليات، وإزاي نخزن كل ده في المتصفح عشان ما يضيعش لو المستخدم أغلق الصفحة.

قبل ما نبدأ، خليني أشرحلك هيكل الملفات ده. كل ملف ليه دور معين:

  • CartContext.jsx - ده الملف الأهم، فيه كل المنطق بتاع السلة: إضافة منتجات، حذف، تحديث الكمية، وحساب الإجماليات
  • CartProvider.jsx - مكون بسيط بيلف التطبيق كله بسياق السلة
  • CartItem.jsx - بطاقة بتمثل كل منتج في السلة مع صورته واسمه وكميته وسعره
  • CartSummary.jsx - ملخص بيعرض الإجماليات: المجموع الفرعي، الشحن، الضرائب، والإجمالي الكلي
  • EmptyCart.jsx - صفحة بتظهر لما السلة تكون فاضية
  • page.js - الصفحة الرئيسية للسلة اللي بتجمع كل المكونات دول

🛒 الخطوة 1: CartContext

الآن نبني السياق لإدارة حالة السلة. السياق (Context) ده أداة في React بتسمحلك تبعت بيانات بين المكونات من غير ما تحتاج تمررها كـ props في كل مرة.

يعني إيه؟ يعني لو عندك مكون الـ Header عايز يعرف عدد المنتجات في السلة، ومكون السلة عايز يضيف منتجات، ومكون صفحة الدفع عايز يعرف المنتجات اللي في السلة - كلهم محتاجين يوصلوا لنفس البيانات. بدل ما نمرر البيانات في كل مكان، بنستخدم Context وبنخزن البيانات في مكان واحد كلهم بيوصلوا ليه.

بنستخدم useReducer هنا بدلاً من useState لأن حالة السلة معقدة شوية - فيها إضافة وحذف وتحديث وحسابات. الـ useReducer بيفرّق منطق التحديث عن المكونات وبيخلي الكود أسهل في الصيانة.

context/CartContext.jsx jsx 
"use client";

import { createContext, useContext, useReducer, useEffect } from 'react';

const CartContext = createContext(null);

// أنواع الإجراءات
const CART_ACTIONS = {
  ADD_ITEM: 'ADD_ITEM',
  REMOVE_ITEM: 'REMOVE_ITEM',
  UPDATE_QUANTITY: 'UPDATE_QUANTITY',
  CLEAR_CART: 'CLEAR_CART',
  LOAD_CART: 'LOAD_CART',
};

//Reducer للسلة
function cartReducer(state, action) {
  switch (action.type) {
    case CART_ACTIONS.ADD_ITEM: {
      const { item, quantity } = action.payload;
      const existingItemIndex = state.items.findIndex(
        (i) => i.id === item.id && i.selectedSize === item.selectedSize && i.selectedColor === item.selectedColor
      );

      if (existingItemIndex > -1) {
        // تحديث الكمية إذا كان المنتج موجوداً
        const updatedItems = [...state.items];
        updatedItems[existingItemIndex].quantity += quantity;
        return { ...state, items: updatedItems };
      }

      // إضافة منتج جديد
      return {
        ...state,
        items: [...state.items, { ...item, quantity }],
      };
    }

    case CART_ACTIONS.REMOVE_ITEM: {
      return {
        ...state,
        items: state.items.filter(
          (i) => !(i.id === action.payload.id && i.selectedSize === action.payload.selectedSize && i.selectedColor === action.payload.selectedColor)
        ),
      };
    }

    case CART_ACTIONS.UPDATE_QUANTITY: {
      const { id, quantity, selectedSize, selectedColor } = action.payload;
      
      if (quantity <= 0) {
        return {
          ...state,
          items: state.items.filter(
            (i) => !(i.id === id && i.selectedSize === selectedSize && i.selectedColor === selectedColor)
          ),
        };
      }

      return {
        ...state,
        items: state.items.map((item) =>
          item.id === id && item.selectedSize === selectedSize && item.selectedColor === selectedColor
            ? { ...item, quantity }
            : item
        ),
      };
    }

    case CART_ACTIONS.CLEAR_CART: {
      return { ...state, items: [] };
    }

    case CART_ACTIONS.LOAD_CART: {
      return { ...state, items: action.payload, isLoading: false };
    }

    default:
      return state;
  }
}

// المزود
export function CartProvider({ children }) {
  const [state, dispatch] = useReducer(cartReducer, {
    items: [],
    isLoading: true,
  });

  // تحميل السلة من localStorage
  useEffect(() => {
    const savedCart = localStorage.getItem('cart');
    if (savedCart) {
      try {
        const parsedCart = JSON.parse(savedCart);
        dispatch({ type: CART_ACTIONS.LOAD_CART, payload: parsedCart });
      } catch (error) {
        console.error('Error loading cart:', error);
        dispatch({ type: CART_ACTIONS.LOAD_CART, payload: [] });
      }
    } else {
      dispatch({ type: CART_ACTIONS.LOAD_CART, payload: [] });
    }
  }, []);

  // حفظ السلة في localStorage
  useEffect(() => {
    if (!state.isLoading) {
      localStorage.setItem('cart', JSON.stringify(state.items));
    }
  }, [state.items, state.isLoading]);

  // إضافة منتج
  const addItem = (item, quantity = 1) => {
    dispatch({
      type: CART_ACTIONS.ADD_ITEM,
      payload: { item, quantity },
    });
  };

  // إزالة منتج
  const removeItem = (id, selectedSize, selectedColor) => {
    dispatch({
      type: CART_ACTIONS.REMOVE_ITEM,
      payload: { id, selectedSize, selectedColor },
    });
  };

  // تحديث الكمية
  const updateQuantity = (id, quantity, selectedSize, selectedColor) => {
    dispatch({
      type: CART_ACTIONS.UPDATE_QUANTITY,
      payload: { id, quantity, selectedSize, selectedColor },
    });
  };

  // تفريغ السلة
  const clearCart = () => {
    dispatch({ type: CART_ACTIONS.CLEAR_CART });
  };

  // حساب الإجماليات
  const getTotals = () => {
    const subtotal = state.items.reduce(
      (sum, item) => sum + item.price * item.quantity,
      0
    );
    const shipping = subtotal > 50 ? 0 : 9.99;
    const tax = subtotal * 0.1; // 10% ضريبة
    const total = subtotal + shipping + tax;

    return {
      subtotal: subtotal.toFixed(2),
      shipping: shipping.toFixed(2),
      tax: tax.toFixed(2),
      total: total.toFixed(2),
      itemCount: state.items.reduce((sum, item) => sum + item.quantity, 0),
    };
  };

  const value = {
    items: state.items,
    isLoading: state.isLoading,
    addItem,
    removeItem,
    updateQuantity,
    clearCart,
    getTotals,
  };

  return (
    <CartContext.Provider value={value}>
      {children}
    </CartContext.Provider>
  );
}

// Hook لاستخدام السلة
export function useCart() {
  const context = useContext(CartContext);
  
  if (!context) {
    throw new Error('useCart must be used within CartProvider');
  }
  
  return context;
}
💡 لماذا useReducer؟

استخدمنا useReducer بدلاً من useState لأنه:

  • مناسب للحالة المعقدة (كثيرة التحديثات)
  • يفصل منطق التحديث عن المكونات
  • يسهل الاختبار والصيانة
  • يوفر أداء أفضل للتحديثات المتكررة

شرح تفصيلي للكود:

خليني أشرحلك كل جزء في الكود ده:

  • CART_ACTIONS - ده object فيه كل الإجراءات اللي ممكن نعملها على السلة: إضافة منتج، حذف منتج، تحديث الكمية، تفريغ السلة، وتحميل السلة من التخزين المحلي
  • cartReducer - ده الـ function اللي بتاخد الحالة الحالية + إجراء وترجع حالة جديدة. زي ما المكان بياخد طلب ويرجع نتيجة
  • addItem - لما المستخدم يضغط "أضف للسلة"، بنستدعي الدالة دي وبتبعت الإجراء للـ reducer
  • removeItem - لما المستخدم يضغط "حذف"، بنحذف المنتج من المصفوفة
  • updateQuantity - لما المستخدم يغير الكمية (+1 أو -1)
  • getTotals - دالة بتحسب كل الإجماليات: المجموع الفرعي، الشحن، الضرائب، والإجمالي الكلي
⚠️ أخطاء شائعة
  • خطأ شائع 1: نسيان التحقق من نفس المقاس واللون عند إضافة منتج. لو المنتج نفسه بس مقاس مختلف، لازم يبقى منتج منفصل في السلة!
  • خطأ شائع 2: استخدام state.items.push() بدلاً من spread operator [...state.items, newItem]. الـ state لازم يبقى immutable (ثابت)، يعني ما نعدّلهوش مباشرة
  • خطأ شائع 3: نسيان معالجة الأخطاء في localStorage. ممكن يحصل خطأ لو المتصفح ما بيدعمش التخزين المحلي

📦 الخطوة 2: مزود السلة

الآن محتاجين نعمل مزود (Provider) بسيط يلف التطبيق كله بسياق السلة. المزود ده بيخلي أي مكون في التطبيق يقدر يستخدم الـ useCart hook.

components/cart/CartProvider.jsx jsx 
"use client";

import { CartProvider as Provider } from '@/context/CartContext';

export default function CartProvider({ children }) {
  return <Provider>{children}</Provider>;
}

ربط المزود بالتطبيق:

الآن لازم نربط الـ CartProvider بالتطبيق كله عشان كل المكونات تقدر توصل للسلة. بنعمل ده في ملف layout.js:

💡 نصيحة مهمة

لاحظ إننا بنلف CartProvider بـ ToastProvider (مش العكس). السبب إننا عايزين الـ Toast يبقى متاح لكل المكوناتكل المكونات. لو حطين CartProvider في الأول، مش هنقدر نستخدم الـ Toast في أخطاء السلة!

🛒 الخطوة 3: صفحة السلة

الآن بنعمل الصفحة الرئيسية للسلة. الصفحة دي بتجمع كل مكونات السلة مع بعض: قائمة المنتجات، ملخص الطلب، وزر إتمام الشراء.

في الصفحة دي بنعمل أشياء مهمة:

  • بنتحقق هل السلة فاضية ولا فيها منتجات
  • بنعرض loading spinner أثناء تحميل السلة
  • بنعرض منتجات السلة في مكون CartItem
  • بنعرض ملخص الطلب في مكون CartSummary
  • بنضيف زر "تفريغ السلة" و"متابعة التسوق"
app/(shop)/cart/page.js jsx 
"use client";

import { useCart } from '@/context/CartContext';
import CartItem from '@/components/cart/CartItem';
import CartSummary from '@/components/cart/CartSummary';
import EmptyCart from '@/components/cart/EmptyCart';
import Container from '@/components/ui/Container';
import Button from '@/components/ui/Button';
import Link from 'next/link';

export default function CartPage() {
  const { items, isLoading, clearCart, getTotals } = useCart();
  const totals = getTotals();

  if (isLoading) {
    return (
      <Container className="py-16">
        <div className="text-center">
          <div className="w-16 h-16 mx-auto border-4 border-indigo-500 border-t-transparent rounded-full animate-spin" />
          <p className="mt-4 text-gray-400">جاري تحميل السلة...</p>
        </div>
      </Container>
    );
  }

  if (items.length === 0) {
    return <EmptyCart />;
  }

  return (
    <Container className="py-8">
      {/* Header */}
      <div className="flex items-center justify-between mb-8">
        <div>
          <h1 className="text-3xl font-bold text-white">سلة المشتريات</h1>
          <p className="text-gray-400 mt-1">
            {totals.itemCount} منتج في السلة
          </p>
        </div>
        <Button
          variant="ghost"
          onClick={clearCart}
          className="text-red-400 hover:text-red-300"
        >
          تفريغ السلة
        </Button>
      </div>

      <div className="grid lg:grid-cols-3 gap-8">
        {/* Cart Items */}
        <div className="lg:col-span-2 space-y-4">
          {items.map((item) => (
            <CartItem key={`${item.id}-${item.selectedSize}-${item.selectedColor}`} item={item} />
          ))}
        </div>

        {/* Summary */}
        <div className="lg:col-span-1">
          <CartSummary totals={totals} />
        </div>
      </div>

      {/* Continue Shopping */}
      <div className="mt-8 text-center">
        <Link href="/products">
          <Button variant="secondary">← متابعة التسوق</Button>
        </Link>
      </div>
    </Container>
  );
}
⚠️ أخطاء شائعة في صفحة السلة
  • خطأ شائع 1: استخدام key={item.id} بس. لو نفس المنتج بس مقاسات مختلفة، الـ key لازم يكون key={`\${item.id}-\${item.selectedSize}-\${item.selectedColor}`}
  • خطأ شائع 2: نسيان التحقق من isLoading. لو المستخدم فتح الصفحة قبل ما السلة تتحمل، هيكون في مشكلة
  • خطأ شائع 3: استخدام items.length === 0 قبل التحقق من isLoading. لازم الأول نتأكد إن التحميل خلص

📝 الخطوة 4: عنصر السلة

الآن بنعمل مكون CartItem اللي بيمثل كل منتج في السلة. كل منتج بيعرض: صورته، اسمه، المقاس واللون (لو موجودين)، الكمية، والسعر.

في المكون ده بنستخدم أشياء مهمة:

  • useCart() للوصول لدوال الحذف وتحديث الكمية
  • أزرار + و - لتحديث الكمية
  • زر الحذف (سلة المهملات)
  • حساب السعر الإجمالي للمنتج = السعر × الكمية
components/cart/CartItem.jsx jsx 
"use client";

import { useCart } from '@/context/CartContext';
import { Trash2, Minus, Plus } from 'lucide-react';
import { cn } from '@/lib/cn';
import Image from 'next/image';

export default function CartItem({ item }) {
  const { removeItem, updateQuantity } = useCart();

  const handleQuantityChange = (newQuantity) => {
    updateQuantity(item.id, newQuantity, item.selectedSize, item.selectedColor);
  };

  return (
    <div className="flex gap-4 p-4 bg-white/5 backdrop-blur-lg rounded-2xl border border-white/10">
      {/* Image */}
      <div className="relative w-24 h-24 flex-shrink-0 rounded-xl overflow-hidden">
        <Image
          src={item.image}
          alt={item.name}
          fill
          className="object-cover"
        />
      </div>

      {/* Info */}
      <div className="flex-1 min-w-0">
        <div className="flex items-start justify-between gap-2">
          <div>
            <h3 className="text-white font-medium truncate">
              {item.name}
            </h3>
            <div className="flex items-center gap-2 mt-1">
              {item.selectedSize && (
                <span className="text-gray-400 text-sm">
                  المقاس: {item.selectedSize}
                </span>
              )}
              {item.selectedColor && (
                <span className="text-gray-400 text-sm">
                  اللون: {item.selectedColor}
                </span>
              )}
            </div>
          </div>

          {/* Delete Button */}
          <button
            onClick={() => removeItem(item.id, item.selectedSize, item.selectedColor)}
            className="p-2 rounded-lg text-gray-400 hover:text-red-400 hover:bg-red-400/10 transition-colors"
          >
            <Trash2 className="w-5 h-5" />
          </button>
        </div>

        {/* Price & Quantity */}
        <div className="flex items-center justify-between mt-4">
          {/* Quantity Selector */}
          <div className="flex items-center gap-2 bg-white/5 rounded-lg">
            <button
              onClick={() => handleQuantityChange(item.quantity - 1)}
              disabled={item.quantity <= 1}
              className={cn(
                "w-8 h-8 flex items-center justify-center rounded-lg transition-colors",
                item.quantity <= 1
                  ? "text-gray-600"
                  : "text-white hover:bg-white/10"
              )}
            >
              <Minus className="w-4 h-4" />
            </button>
            <span className="w-8 text-center text-white font-medium">
              {item.quantity}
            </span>
            <button
              onClick={() => handleQuantityChange(item.quantity + 1)}
              className="w-8 h-8 flex items-center justify-center rounded-lg text-white hover:bg-white/10 transition-colors"
            >
              <Plus className="w-4 h-4" />
            </button>
          </div>

          {/* Price */}
          <div className="text-left">
            <p className="text-white font-bold">
              ${(item.price * item.quantity).toFixed(2)}
            </p>
            {item.quantity > 1 && (
              <p className="text-gray-400 text-sm">
                ${item.price.toFixed(2)} للواحدة
              </p>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}

شرح مكون CartItem:

خليني أشرحلك أهم أجزاء المكون ده:

  • handleQuantityChange: لما المستخدم يضغط + أو -، الدالة دي بتستدعي updateQuantity بالكمية الجديدة
  • disabled={item.quantity <= 1}: لو الكمية 1، زر الناقص (-) بيتعطّل عشان ما ننزلش الكمية تحت الصفر
  • `${(item.price * item.quantity).toFixed(2)}`: بنحسب السعر الإجمالي للمنتج = السعر × الكمية. الـ toFixed(2) بيخلّي الرقم يعرض بعشريتين (مثلاً 99.99)
  • {item.quantity > 1 && ...}: لو الكمية أكبر من 1، بنعرض السعر للواحدة تحت السعر الإجمالي
💡 نصيحة للتصميم

لاحظ إننا بنستخدم truncate على اسم المنتج. لو الاسم طويل أوي، هيتعمله نقاط (...) بدل ما يكسر التصميم. دي من أفضل الممارسات في تصميم البطاقات!

        </div>
        <div className="flex items-center gap-3 text-gray-400 text-sm">
          <CreditCard className="w-5 h-5 text-purple-400" />
          <span>طرق دفع متعددة</span>
        </div>
      </div>

      {/* Free Shipping Notice */}
      {parseFloat(totals.subtotal) < 50 && (
        <div className="mt-4 p-3 bg-green-500/10 border border-green-500/20 rounded-xl">
          <p className="text-green-400 text-sm text-center">
            أضف ${(50 - parseFloat(totals.subtotal)).toFixed(2)} للحصول على شحن مجاني!
          </p>
        </div>
      )}
    </Card>
  );
}

شرح مكون CartSummary:

  • sticky top-24: الـ summary بيفضل ثابت في الشاشة لو المستخدم scrolledDown. ده تجربة مستخدم ممتازة!
  • الشحن المجاني: لو المجموع الفرعي أكبر من 50 دولار، الشحن بيبقى مجاني. لو أقل، بنعرض رسالة بتقول المستخدم محتاج يضيف قد إيه عشان يحصل على الشحن المجاني
  • Trust Badges: أيقونات بتبين للمستخدم إن المتجر آمن وموثوق. دي بتزود ثقة المستخدم وتخليه يكمل عملية الشراء
