🗄️ الخطوة 1: جدول Orders في Supabase

أولاً محتاجين نعمل جدول في Supabase لتخزين الطلبات. روح على Supabase Dashboard واذهب إلى SQL Editor وأدخل الكود ده:

supabase/migrations/create_orders.sql sql 
-- جدول الطلبات الرئيسي
CREATE TABLE orders (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'pending'
    CHECK (status IN ('pending', 'confirmed', 'processing', 'shipped', 'delivered', 'cancelled')),
  payment_status TEXT NOT NULL DEFAULT 'pending'
    CHECK (payment_status IN ('pending', 'paid', 'failed', 'refunded')),
  payment_method TEXT NOT NULL DEFAULT 'credit_card',
  shipping_address JSONB NOT NULL,
  subtotal DECIMAL(10, 2) NOT NULL,
  shipping_cost DECIMAL(10, 2) DEFAULT 0,
  tax DECIMAL(10, 2) DEFAULT 0,
  total DECIMAL(10, 2) NOT NULL,
  notes TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- جدول عناصر الطلب
CREATE TABLE order_items (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  order_id UUID NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
  product_id UUID NOT NULL REFERENCES products(id),
  quantity INTEGER NOT NULL CHECK (quantity > 0),
  price DECIMAL(10, 2) NOT NULL,
  selected_size TEXT,
  selected_color TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- الفهارس لتحسين الأداء
CREATE INDEX idx_orders_user_id ON orders(user_id);
CREATE INDEX idx_orders_status ON orders(status);
CREATE INDEX idx_orders_created_at ON orders(created_at DESC);
CREATE INDEX idx_order_items_order_id ON order_items(order_id);

-- تفعيل Row Level Security
ALTER TABLE orders ENABLE ROW LEVEL SECURITY;
ALTER TABLE order_items ENABLE ROW LEVEL SECURITY;

-- سياسة للقراءة: المستخدم يقدر يشوف طلباته بس
CREATE POLICY "Users can view their own orders"
  ON orders FOR SELECT
  USING (auth.uid()::text = user_id);

-- سياسة للإضافة: المستخدم يقدر يضيف طلبات
CREATE POLICY "Users can create orders"
  ON orders FOR INSERT
  WITH CHECK (auth.uid()::text = user_id);

-- سياسة للقراءة: المستخدم يقدر يشوف عناصر طلباته
CREATE POLICY "Users can view their order items"
  ON order_items FOR SELECT
  USING (
    order_id IN (
      SELECT id FROM orders WHERE auth.uid()::text = user_id
    )
  );

-- سياسة للإضافة: المستخدم يقدر يضيف عناصر لطلباته
CREATE POLICY "Users can create order items"
  ON order_items FOR INSERT
  WITH CHECK (
    order_id IN (
      SELECT id FROM orders WHERE auth.uid()::text = user_id
    )
  );

-- دالة لتحديث updated_at تلقائياً
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ language 'plpgsql';

-- 트리거 لتحديث updated_at
CREATE TRIGGER update_orders_updated_at
  BEFORE UPDATE ON orders
  FOR EACH ROW
  EXECUTE FUNCTION update_updated_at_column();
💡 ملاحظة مهمة
الـ UUID في orders هو مرجع المستخدم من Clerk. لاحظ أننا نستخدم auth.uid()::text لتحويله إلى نص. في بعض الحالات ممكن تحتاج تعدل الـ RLS policies حسب إعدادات Clerk بتاعتك.

📦 الخطوة 2: خدمات الطلبات (Order Service)

بنينا مكون services مخصص لإدارة الطلبات. المكون ده بيشتغل كطبقة وسيطة بين الـ API والـ Supabase:

services/orderService.js javascript 
import { supabase } from '@/lib/supabase';

/**
 * إنشاء طلب جديد
 * @param {Object} orderData - بيانات الطلب
 * @returns {Object} - الطلب المنشأ
 */
export async function createOrder(orderData) {
  const {
    userId,
    items,
    shippingAddress,
    paymentMethod,
    subtotal,
    shippingCost,
    tax,
    total,
    notes
  } = orderData;

  // إنشاء الطلب الرئيسي
  const { data: order, error: orderError } = await supabase
    .from('orders')
    .insert([
      {
        user_id: userId,
        status: 'pending',
        payment_status: paymentMethod === 'cod' ? 'pending' : 'paid',
        payment_method: paymentMethod,
        shipping_address: shippingAddress,
        subtotal: parseFloat(subtotal),
        shipping_cost: parseFloat(shippingCost),
        tax: parseFloat(tax),
        total: parseFloat(total),
        notes: notes || null,
      },
    ])
    .select()
    .single();

  if (orderError) {
    console.error('Error creating order:', orderError);
    throw new Error('فشل في إنشاء الطلب');
  }

  // إضافة عناصر الطلب
  const orderItems = items.map((item) => ({
    order_id: order.id,
    product_id: item.productId || item.id,
    quantity: item.quantity,
    price: parseFloat(item.price),
    selected_size: item.selectedSize || null,
    selected_color: item.selectedColor || null,
  }));

  const { error: itemsError } = await supabase
    .from('order_items')
    .insert(orderItems);

  if (itemsError) {
    console.error('Error creating order items:', itemsError);
    throw new Error('فشل في إضافة عناصر الطلب');
  }

  return order;
}

/**
 * جلب طلبات المستخدم
 * @param {string} userId - معرف المستخدم
 * @param {Object} options - خيارات إضافية
 * @returns {Array} - قائمة الطلبات
 */
export async function getUserOrders(userId, options = {}) {
  const { status, limit = 10, offset = 0 } = options;

  let query = supabase
    .from('orders')
    .select(`
      *,
      order_items (
        *,
        products (
          id,
          name,
          image
        )
      )
    `)
    .eq('user_id', userId)
    .order('created_at', { ascending: false })
    .range(offset, offset + limit - 1);

  // تصفية حسب الحالة إذا كانت محددة
  if (status) {
    query = query.eq('status', status);
  }

  const { data, error } = await query;

  if (error) {
    console.error('Error fetching orders:', error);
    throw new Error('فشل في جلب الطلبات');
  }

  return data;
}

/**
 * جلب تفاصيل طلب محدد
 * @param {string} orderId - معرف الطلب
 * @param {string} userId - معرف المستخدم للتحقق من الصلاحية
 * @returns {Object} - تفاصيل الطلب
 */
export async function getOrderById(orderId, userId) {
  const { data, error } = await supabase
    .from('orders')
    .select(`
      *,
      order_items (
        *,
        products (
          id,
          name,
          image,
          description
        )
      )
    `)
    .eq('id', orderId)
    .eq('user_id', userId)
    .single();

  if (error) {
    console.error('Error fetching order:', error);
    throw new Error('الطلب غير موجود');
  }

  return data;
}

/**
 * تحديث حالة الطلب (للadmin فقط)
 * @param {string} orderId - معرف الطلب
 * @param {string} status - الحالة الجديدة
 * @returns {Object} - الطلب المحدث
 */
export async function updateOrderStatus(orderId, status) {
  const { data, error } = await supabase
    .from('orders')
    .update({ status, updated_at: new Date().toISOString() })
    .eq('id', orderId)
    .select()
    .single();

  if (error) {
    console.error('Error updating order status:', error);
    throw new Error('فشل في تحديث حالة الطلب');
  }

  return data;
}

/**
 * إلغاء طلب
 * @param {string} orderId - معرف الطلب
 * @param {string} userId - معرف المستخدم
 * @returns {Object} - الطلب المُلغى
 */
export async function cancelOrder(orderId, userId) {
  // التحقق من أن الطلب يخص المستخدم وأن حالته تسمح بالإلغاء
  const { data: order, error: fetchError } = await supabase
    .from('orders')
    .select('status')
    .eq('id', orderId)
    .eq('user_id', userId)
    .single();

  if (fetchError || !order) {
    throw new Error('الطلب غير موجود');
  }

  if (!['pending', 'confirmed'].includes(order.status)) {
    throw new Error('لا يمكن إلغاء الطلب في هذه الحالة');
  }

  const { data, error } = await supabase
    .from('orders')
    .update({
      status: 'cancelled',
      payment_status: 'refunded',
      updated_at: new Date().toISOString()
    })
    .eq('id', orderId)
    .select()
    .single();

  if (error) {
    throw new Error('فشل في إلغاء الطلب');
  }

  return data;
}

🌐 الخطوة 3: API Routes للطلبات

بنينا API Routes للتعامل مع الطلبات من خلال REST API:

app/api/orders/route.js javascript 
import { NextResponse } from 'next/server';
import { createOrder, getUserOrders } from '@/services/orderService';

// POST: إنشاء طلب جديد
export async function POST(request) {
  try {
    const body = await request.json();
    const {
      userId,
      items,
      shippingAddress,
      paymentMethod,
      subtotal,
      shippingCost,
      tax,
      total,
      notes
    } = body;

    // التحقق من البيانات المطلوبة
    if (!userId || !items || items.length === 0 || !shippingAddress) {
      return NextResponse.json(
        { error: 'بيانات الطلب غير مكتملة' },
        { status: 400 }
      );
    }

    // التحقق من وجود المنتجات
    if (items.some(item => !item.productId && !item.id)) {
      return NextResponse.json(
        { error: 'جميع العناصر يجب أن تحتوي على معرف المنتج' },
        { status: 400 }
      );
    }

    const order = await createOrder({
      userId,
      items,
      shippingAddress,
      paymentMethod,
      subtotal,
      shippingCost,
      tax,
      total,
      notes
    });

    return NextResponse.json(order, { status: 201 });
  } catch (error) {
    console.error('Error creating order:', error);
    return NextResponse.json(
      { error: error.message || 'فشل في إنشاء الطلب' },
      { status: 500 }
    );
  }
}

// GET: جلب طلبات المستخدم
export async function GET(request) {
  try {
    const { searchParams } = new URL(request.url);
    const userId = searchParams.get('userId');
    const status = searchParams.get('status');
    const limit = parseInt(searchParams.get('limit')) || 10;
    const offset = parseInt(searchParams.get('offset')) || 0;

    if (!userId) {
      return NextResponse.json(
        { error: 'معرف المستخدم مطلوب' },
        { status: 400 }
      );
    }

    const orders = await getUserOrders(userId, { status, limit, offset });

    return NextResponse.json(orders);
  } catch (error) {
    console.error('Error fetching orders:', error);
    return NextResponse.json(
      { error: error.message || 'فشل في جلب الطلبات' },
      { status: 500 }
    );
  }
}

🔗 الخطوة 7: API Route لتفاصيل الطلب

app/api/orders/[id]/route.js javascript 
import { NextResponse } from 'next/server';
import { getOrderById, updateOrderStatus } from '@/services/orderService';

// GET: جلب تفاصيل طلب محدد
export async function GET(request, { params }) {
  try {
    const { id } = params;
    const { searchParams } = new URL(request.url);
    const userId = searchParams.get('userId');

    if (!userId) {
      return NextResponse.json(
        { error: 'معرف المستخدم مطلوب' },
        { status: 400 }
      );
    }

    const order = await getOrderById(id, userId);

    return NextResponse.json(order);
  } catch (error) {
    console.error('Error fetching order:', error);
    return NextResponse.json(
      { error: error.message || 'الطلب غير موجود' },
      { status: 404 }
    );
  }
}

// PATCH: تحديث حالة الطلب (للadmin فقط)
export async function PATCH(request, { params }) {
  try {
    const { id } = params;
    const body = await request.json();
    const { status } = body;

    // التحقق من الحالة المسموحة
    const validStatuses = ['pending', 'confirmed', 'processing', 'shipped', 'delivered', 'cancelled'];
    if (!validStatuses.includes(status)) {
      return NextResponse.json(
        { error: 'حالة غير صالحة' },
        { status: 400 }
      );
    }

    // هنا يجب التحقق من صلاحيات المستخدم (admin)
    // يمكنك إضافة middleware للتحقق من Clerk

    const order = await updateOrderStatus(id, status);

    return NextResponse.json(order);
  } catch (error) {
    console.error('Error updating order:', error);
    return NextResponse.json(
      { error: error.message || 'فشل في تحديث الطلب' },
      { status: 500 }
    );
  }
}
