❤️ الخطوة 1: WishlistContext

بنينا Context مخصص لإدارة حالة المفضلة. السياق ده بيستخدم localStorage لتخزين المنتجات المفضلة حتى لو المستخدم أغلق المتصفح:

context/WishlistContext.js javascript 
"use client";

import { createContext, useContext, useState, useEffect } from 'react';

const WishlistContext = createContext(null);

const STORAGE_KEY = 'market_wishlist';

export function WishlistProvider({ children }) {
  const [items, setItems] = useState([]);
  const [isLoaded, setIsLoaded] = useState(false);

  // تحميل المفضلة من localStorage عند بدء التشغيل
  useEffect(() => {
    try {
      const saved = localStorage.getItem(STORAGE_KEY);
      if (saved) {
        setItems(JSON.parse(saved));
      }
    } catch (error) {
      console.error('Error loading wishlist:', error);
    } finally {
      setIsLoaded(true);
    }
  }, []);

  // حفظ المفضلة في localStorage عند التغيير
  useEffect(() => {
    if (isLoaded) {
      try {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(items));
      } catch (error) {
        console.error('Error saving wishlist:', error);
      }
    }
  }, [items, isLoaded]);

  /**
   * إضافة أو إزالة منتج من المفضلة
   * @param {Object} product - المنتج
   */
  const toggleItem = (product) => {
    setItems((prev) => {
      const exists = prev.some((item) => item.id === product.id);
      if (exists) {
        return prev.filter((item) => item.id !== product.id);
      } else {
        return [...prev, {
          id: product.id,
          name: product.name,
          price: product.price,
          image: product.image,
          slug: product.slug,
          addedAt: new Date().toISOString(),
        }];
      }
    });
  };

  /**
   * إزالة منتج من المفضلة
   * @param {string} productId - معرف المنتج
   */
  const removeItem = (productId) => {
    setItems((prev) => prev.filter((item) => item.id !== productId));
  };

  /**
   * التحقق من وجود منتج في المفضلة
   * @param {string} productId - معرف المنتج
   * @returns {boolean} - هل المنتج في المفضلة
   */
  const isInWishlist = (productId) => {
    return items.some((item) => item.id === productId);
  };

  /**
   * مسح جميع العناصر من المfavorite
   */
  const clearWishlist = () => {
    setItems([]);
  };

  /**
   * عدد العناصر في المفضلة
   */
  const itemCount = items.length;

  const value = {
    items,
    isLoaded,
    toggleItem,
    removeItem,
    isInWishlist,
    clearWishlist,
    itemCount,
  };

  return (
    
      {children}
    
  );
}

/**
 * Hook لاستخدام سياق المفضلة
 */
export function useWishlist() {
  const context = useContext(WishlistContext);
  if (!context) {
    throw new Error('useWishlist must be used within a WishlistProvider');
  }
  return context;
}
💡 لماذا localStorage؟
نستخدم localStorage لتخزين المفضلة محلياً حتى لو المستخدم غير مسجل دخول. هذا يعطي تجربة مستخدم أفضل لأن المستخدم يقدر يحفظ منتجاته المفضلة بدون تسجيل دخول. إذا أردت مزامنة المفضلة مع حساب المستخدم، يمكنك استخدام Supabase.
