💡 مقدمة للدرس

أهلاً بيك في الدرس الستاشر! في الدرس ده هنبني نظام المفضلة. نظام المفضلة ده من أكتر المميزات المحبوبة في المتاجر الإلكترونية - بيسمح للمستخدم يحفظ المنتجات اللي عجبته عشان يرجع ليها لاحقاً.

في الدرس ده هنتعلم إزاي نستخدم localStorage عشان نحفظ المفضلة محلياً في المتصفح. يعني حتى لو المستخدم ما عندهوش حساب في المتجر، يقدر يحفظ منتجاته المفضلة!

❤️ الخطوة 1: WishlistContext

بنينا Context مخصص لإدارة حالة المفضلة. السياق ده بيستخدم localStorage لتخزين المنتجات المفضلة حتى لو المستخدم أغلق المتصفح:

context/WishlistContext.js javascript 
"use client";

import { createContext, useContext, useState, useEffect } from 'react';

const WishlistContext = createContext(null);

const STORAGE_KEY = 'market_wishlist';

export function WishlistProvider({ children }) {
  const [items, setItems] = useState([]);
  const [isLoaded, setIsLoaded] = useState(false);

  // تحميل المفضلة من localStorage عند بدء التشغيل
  useEffect(() => {
    try {
      const saved = localStorage.getItem(STORAGE_KEY);
      if (saved) {
        setItems(JSON.parse(saved));
      }
    } catch (error) {
      console.error('Error loading wishlist:', error);
    } finally {
      setIsLoaded(true);
    }
  }, []);

  // حفظ المفضلة في localStorage عند التغيير
  useEffect(() => {
    if (isLoaded) {
      try {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(items));
      } catch (error) {
        console.error('Error saving wishlist:', error);
      }
    }
  }, [items, isLoaded]);

  /**
   * إضافة أو إزالة منتج من المفضلة
   * @param {Object} product - المنتج
   */
  const toggleItem = (product) => {
    setItems((prev) => {
      const exists = prev.some((item) => item.id === product.id);
      if (exists) {
        return prev.filter((item) => item.id !== product.id);
      } else {
        return [...prev, {
          id: product.id,
          name: product.name,
          price: product.price,
          image: product.image,
          slug: product.slug,
          addedAt: new Date().toISOString(),
        }];
      }
    });
  };

  /**
   * إزالة منتج من المفضلة
   * @param {string} productId - معرف المنتج
   */
  const removeItem = (productId) => {
    setItems((prev) => prev.filter((item) => item.id !== productId));
  };

  /**
   * التحقق من وجود منتج في المفضلة
   * @param {string} productId - معرف المنتج
   * @returns {boolean} - هل المنتج في المفضلة
   */
  const isInWishlist = (productId) => {
    return items.some((item) => item.id === productId);
  };

  /**
   * مسح جميع العناصر من المfavorite
   */
  const clearWishlist = () => {
    setItems([]);
  };

  /**
   * عدد العناصر في المفضلة
   */
  const itemCount = items.length;

  const value = {
    items,
    isLoaded,
    toggleItem,
    removeItem,
    isInWishlist,
    clearWishlist,
    itemCount,
  };

  return (
    <WishlistContext.Provider value={value}>
      {children}
    </WishlistContext.Provider>
  );
}

/**
 * Hook لاستخدام سياق المفضلة
 */
export function useWishlist() {
  const context = useContext(WishlistContext);
  if (!context) {
    throw new Error('useWishlist must be used within a WishlistProvider');
  }
  return context;
}
💡 لماذا localStorage؟

نستخدم localStorage لتخزين المفضلة محلياً حتى لو المستخدم غير مسجل دخول. هذا يعطي تجربة مستخدم أفضل لأن المستخدم يقدر يحفظ منتجاته المفضلة بدون تسجيل دخول. إذا أردت مزامنة المفضلة مع حساب المستخدم، يمكنك استخدام Supabase.

"use client";

import { useWishlist } from '@/context/WishlistContext';
import Link from 'next/link';
import { Heart, ShoppingBag, Trash2, ArrowRight } from 'lucide-react';
import { motion, AnimatePresence } from 'framer-motion';
import Container from '@/components/ui/Container';
import Card from '@/components/ui/Card';
import Button from '@/components/ui/Button';
import ProductCard from '@/components/products/ProductCard';

export default function WishlistPage() {
  const { items, removeItem, clearWishlist, itemCount, isLoaded } = useWishlist();

  if (!isLoaded) {
    return (
      <Container className="py-16">
        <div className="text-center">
          <div className="w-16 h-16 mx-auto border-4 border-indigo-500 border-t-transparent rounded-full animate-spin" />
          <p className="mt-4 text-gray-400">جاري التحميل...</p>
        </div>
      </Container>
    );
  }

  return (
    <Container className="py-16">
      <div className="max-w-6xl mx-auto">
        {/* Header */}
        <div className="flex flex-col sm:flex-row sm:items-center justify-between gap-4 mb-8">
          <div>
            <h1 className="text-3xl font-bold text-white mb-2">
              <Heart className="inline-block w-8 h-8 ml-3 text-pink-500" />
              المفضلة
            </h1>
            <p className="text-gray-400">
              {itemCount > 0 
                ? `لديك ${itemCount} منتج في المفضلة`
                : 'لم تобав أي منتج بعد'
              }
            </p>
          </div>

          {itemCount > 0 && (
            <Button
              variant="secondary"
              onClick={clearWishlist}
              className="text-red-400 hover:text-red-300"
            >
              <Trash2 className="w-5 h-5 ml-2" />
              مسح الكل
            </Button>
          )}
        </div>

        {/* Empty State */}
        {itemCount === 0 ? (
          <Card variant="glass" className="text-center py-16">
            <motion.div
              initial={{ scale: 0 }}
              animate={{ scale: 1 }}
              transition={{ type: "spring", stiffness: 200, damping: 15 }}
            >
              <Heart className="w-24 h-24 text-gray-600 mx-auto mb-6" />
            </motion.div>
            <h2 className="text-2xl font-bold text-white mb-4">
              المفضلة فارغة
            </h2>
            <p className="text-gray-400 mb-8 max-w-md mx-auto">
              أضف المنتجات التي تحبها عن طريق الضغط على أيقونة القلب
            </p>
            <Link href="/products">
              <Button>
                <ShoppingBag className="w-5 h-5 ml-2" />
                تصفح المنتجات
                <ArrowRight className="w-5 h-5 mr-2" />
              </Button>
            </Link>
          </Card>
        ) : (
          <>
            {/* Wishlist Grid */}
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
              <AnimatePresence>
                {items.map((item) => (
                  <motion.div
                    key={item.id}
                    layout
                    initial={{ opacity: 0, scale: 0.8 }}
                    animate={{ opacity: 1, scale: 1 }}
                    exit={{ opacity: 0, scale: 0.8 }}
                    transition={{ duration: 0.3 }}
                  >
                    <ProductCard
                      product={item}
                      isInWishlist={true}
                      onToggleWishlist={() => removeItem(item.id)}
                    />
                  </motion.div>
                ))}
              </AnimatePresence>
            </div>

            {/* Continue Shopping */}
            <div className="mt-12 text-center">
              <Link href="/products">
                <Button variant="secondary">
                  <ArrowRight className="w-5 h-5 ml-2" />
                  متابعة التسوق
                </Button>
              </Link>
            </div>
          
        )}
      </div>
    </Container>
  );
}

🛒 الخطوة 3: زر القلب في ProductCard

components/products/ProductCard.jsx jsx 
"use client";

import { useState } from 'react';
import Link from 'next/link';
import Image from 'next/image';
import { Heart, ShoppingCart, Star, Eye } from 'lucide-react';
import { motion } from 'framer-motion';
import { useWishlist } from '@/context/WishlistContext';
import { useCart } from '@/context/CartContext';
import Button from '@/components/ui/Button';
import { cn } from '@/lib/cn';

export default function ProductCard({ product, isInWishlist: externalIsInWishlist, onToggleWishlist }) {
  const { isInWishlist: checkIsInWishlist, toggleItem } = useWishlist();
  const { addItem } = useCart();
  const [isHovered, setIsHovered] = useState(false);

  // استخدام الحالة الخارجية إذا كانت محددة، وإلا نستخدم Context
  const isInWishlist = externalIsInWishlist !== undefined 
    ? externalIsInWishlist 
    : checkIsInWishlist(product.id);

  const handleToggleWishlist = (e) => {
    e.preventDefault();
    e.stopPropagation();
    
    if (onToggleWishlist) {
      onToggleWishlist();
    } else {
      toggleItem(product);
    }
  };

  const handleAddToCart = (e) => {
    e.preventDefault();
    e.stopPropagation();
    addItem(product);
  };

  return (
    <motion.div
      onHoverStart={() => setIsHovered(true)}
      onHoverEnd={() => setIsHovered(false)}
      className="group"
    >
      <Link href={`/products/${product.slug || product.id}`}>
        <div className="relative bg-white/5 backdrop-blur-lg rounded-2xl border border-white/10 overflow-hidden transition-all duration-300 hover:border-indigo-500/50 hover:shadow-lg hover:shadow-indigo-500/10">
          
          {/* Image Container */}
          <div className="relative aspect-square overflow-hidden bg-gray-900">
            {product.image ? (
              <Image
                src={product.image}
                alt={product.name}
                fill
                className="object-cover transition-transform duration-500 group-hover:scale-110"
              />
            ) : (
              <div className="w-full h-full flex items-center justify-center text-gray-600">
                <ShoppingCart className="w-16 h-16" />
              </div>
            )}

            {/* Overlay */}
            <div className={cn(
              "absolute inset-0 bg-black/40 flex items-center justify-center gap-4 transition-opacity duration-300",
              isHovered ? "opacity-100" : "opacity-0"
            )}>
              <Button
                size="sm"
                onClick={handleAddToCart}
                className="rounded-full"
              >
                <ShoppingCart className="w-5 h-5" />
              </Button>
              <Button
                size="sm"
                variant="secondary"
                className="rounded-full"
              >
                <Eye className="w-5 h-5" />
              </Button>
            </div>

            {/* Wishlist Button */}
            <motion.button
              onClick={handleToggleWishlist}
              className={cn(
                "absolute top-3 right-3 w-10 h-10 rounded-full flex items-center justify-center transition-all duration-300",
                isInWishlist
                  ? "bg-pink-500 text-white"
                  : "bg-black/50 text-white hover:bg-pink-500"
              )}
              whileTap={{ scale: 0.8 }}
            >
              <Heart
                className={cn(
                  "w-5 h-5 transition-all duration-300",
                  isInWishlist && "fill-current"
                )}
              />
            </motion.button>

            {/* Sale Badge */}
            {product.originalPrice && product.originalPrice > product.price && (
              <div className="absolute top-3 left-3 px-3 py-1 bg-red-500 text-white text-sm font-bold rounded-full">
                -{Math.round((1 - product.price / product.originalPrice) * 100)}%
              </div>
            )}
          </div>

          {/* Content */}
          <div className="p-4">
            {/* Category */}
            {product.category && (
              <p className="text-indigo-400 text-sm mb-1">{product.category}</p>
            )}

            {/* Name */}
            <h3 className="text-white font-medium mb-2 line-clamp-2 group-hover:text-indigo-400 transition-colors">
              {product.name}
            </h3>

            {/* Rating */}
            {product.rating && (
              <div className="flex items-center gap-1 mb-2">
                {[...Array(5)].map((_, i) => (
                  <Star
                    key={i}
                    className={cn(
                      "w-4 h-4",
                      i < product.rating
                        ? "text-yellow-400 fill-current"
                        : "text-gray-600"
                    )}
                  />
                ))}
                {product.reviewCount && (
                  <span className="text-gray-400 text-sm mr-2">
                    ({product.reviewCount})
                  </span>
                )}
              </div>
            )}

            {/* Price */}
            <div className="flex items-center gap-2">
              <span className="text-white font-bold text-lg">
                ${product.price.toFixed(2)}
              </span>
              {product.originalPrice && product.originalPrice > product.price && (
                <span className="text-gray-500 line-through text-sm">
                  ${product.originalPrice.toFixed(2)}
                </span>
              )}
            </div>
          </div>
        </div>
      </Link>
    </motion.div>
  );
}

"use client";

import { useState } from 'react';

                </button>

                  {userMenuOpen && (
                    <motion.div
                      initial={{ opacity: 0, y: 10 }}
                      animate={{ opacity: 1, y: 0 }}
                      exit={{ opacity: 0, y: 10 }}
                      className="absolute left-0 mt-2 w-48 bg-gray-900 border border-white/10 rounded-xl shadow-xl overflow-hidden"

                      >
                        <Package className="w-5 h-5" />

                      </Link>
                      <Link
                        href="/settings"
                        className="flex items-center gap-3 px-4 py-3 text-gray-300 hover:bg-white/10 transition-colors"
                        onClick={() => setUserMenuOpen(false)}
                      >
                        <Settings className="w-5 h-5" />
                        الإعدادات
                      </Link>
                      <hr className="border-white/10" />
                      <button
                        onClick={() => {
                          setUserMenuOpen(false);
                          // Clerk signOut
                        }}
                        className="flex items-center gap-3 px-4 py-3 text-red-400 hover:bg-white/10 transition-colors w-full"

                        <LogOut className="w-5 h-5" />
                        تسجيل الخروج
                      </button>
                    </motion.div>
                  )}

              </div>
            ) : (
              <Link href="/sign-in">
                <Button variant="secondary" size="sm">
                  تسجيل الدخول
                </Button>
              </Link>
            )}

            {/* Mobile Menu Toggle */}
            <button
              className="md:hidden p-2 text-gray-300 hover:text-white transition-colors"
              onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
            >
              {mobileMenuOpen ? (
                <X className="w-5 h-5" />
              ) : (
                <Menu className="w-5 h-5" />
              )}
            </button>
          </div>
        </div>
      </div>

      {/* Mobile Menu */}
      <AnimatePresence>
        {mobileMenuOpen && (
          <motion.div
            initial={{ height: 0, opacity: 0 }}
            animate={{ height: 'auto', opacity: 1 }}
            exit={{ height: 0, opacity: 0 }}
            className="md:hidden overflow-hidden bg-gray-900 border-t border-white/10"
          >
            <nav className="container mx-auto px-4 py-4 space-y-2">
              {navLinks.map((link) => (
                <Link
                  key={link.href}
                  href={link.href}
                  className="block px-4 py-3 text-gray-300 hover:bg-white/10 rounded-xl transition-colors"
                  onClick={() => setMobileMenuOpen(false)}
                >
                  {link.label}
                </Link>
              ))}
            </nav>
          </motion.div>
        )}
      </AnimatePresence>
    </header>
  );
}
